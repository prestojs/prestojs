Interface Class<T>

Utility to type a generic class type, eg.

NOTE: If not dealing with generics just use typeof <class> instead

class Base<T> {
  value?: T;
}
class Concrete<T> extends Base<T> {}
function factory<T>(a: Class<Base<T>>): Base<T> {
  return new a();
}
factory(Concrete);
